122void executeRequestAndGetResult(std::unique_ptr<T> _trait,
123 const XpertRequestResult& _xpertRequestResult,
124 std::unique_ptr<U>& _responsePointer)
125{
126 // Make the computing request and response.
127 Core::ComputingRequest computingRequest { "", *_xpertRequestResult.getDrugModel(), *_xpertRequestResult.getTreatment(), move(_trait)};
128 std::unique_ptr<Core::ComputingResponse> computingResponse = std::make_unique<Core::ComputingResponse>("");
129
130 // Start the computation in tucuxi-core.
131 Core::IComputingService* computingComponent = dynamic_cast<Core::IComputingService*>(Core::ComputingComponent::createComponent());
132 Core::ComputingStatus result = computingComponent->compute(computingRequest, computingResponse);
133
134 // If the computation failed, set to nullptr and leave.
135 if (result != Core::ComputingStatus::Ok) {
136 _responsePointer = nullptr;
137 return;
138 }
139
140 // Acquire the data and put it in the unique pointer as the response.
141 U* dataPointer = dynamic_cast<U*>(computingResponse->getUniquePointerData().release());
142 _responsePointer = std::unique_ptr<U>(dataPointer);
143}
